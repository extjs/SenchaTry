Setting animations when an Ext.Msg component is shown or hidden in Sencha Touch 2

The following example shows how you can add an animation when an Ext.Msg component is shown or hidden in Sencha Touch 2 by passing the showAnimation and/or hideAnimation config option to the Ext.Msg.show() method.

Tags:

  • Sencha Touch
  • Sencha Touch 2.0.0
[example]
© 2012 Sencha Inc. 
/*global Ext:false */
Ext.application({
    launch: function () {
        Ext.create('Ext.Panel', {
            fullscreen: true,
            items: [{
                xtype: 'titlebar',
                docked: 'top',
                items: [{
                    text: 'slideIn/slideOut',
                    handler: function () {
                        Ext.Msg.show({
                            title:'slide animations',
                            showAnimation: 'slideIn',
                            hideAnimation: 'slideOut'
                        }); // show()
                    } // handler
                }, {
                    text: 'fadeIn/fadeOut',
                    handler: function () {
                        Ext.Msg.show({
                            title:'fade animations',
                            showAnimation: 'fadeIn',
                            hideAnimation: 'fadeOut'
                        }); // show()
                    } // handler
                }] // items
            }] // items
        }); // create()
    } // launch
}); // application()
