Setting a title bar UI theme on an Ext.TitleBar container in Sencha Touch 2

The following example shows how you can set the title bar UI theme on an Ext.TitleBar container in Sencha Touch 2 by setting the ui config to "dark", "light", or "neutral".

/*global Ext:false */
Ext.application({
    launch: function () {
        Ext.create('Ext.Panel', {
            fullscreen: true,
            items: [{
                docked: 'top',
                title: 'ui: \'dark\' (default)',
                ui: 'dark',
                xtype: 'titlebar'
            }, {
                docked: 'top',
                title: 'ui: \'light\'',
                ui: 'light',
                xtype: 'titlebar'
            }, {
                docked: 'top',
                title: 'ui: \'neutral\'',
                ui: 'neutral',
                xtype: 'titlebar'
            }] // items
        }); // create()
    } // launch
}); // application()
