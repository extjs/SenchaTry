Creating a custom formatting function on an Ext.XTemplate in Sencha Touch 2

The following example shows how you can specify a custom formatting function for an Ext.XTemplate instance in Sencha Touch 2 by passing an object containing your formatting functions to the Ext.XTemplate constructor method, or by using the values object.

/*global Ext:false */
Ext.application({
    launch: function () {
        var myStore = Ext.create('Ext.data.Store', {
            data: [{
                fname: 'Stratton',
                lname: 'Sclavos',
                role: 'Executive Chairman'
            }, {
                fname: 'Michael',
                lname: 'Mullany',
                role: 'CEO'
            }, {
                fname: 'Ted',
                lname: 'Driscoll',
                role: 'Vice President Worldwide Sales'
            }, {
                fname: 'Abraham',
                lname: 'Elias',
                role: 'Chief Technical Officer'
            }, {
                fname: 'Jeff',
                lname: 'Hartley',
                role: 'Vice President of Services and Training'
            }, {
                fname: 'Adam',
                lname: 'Mishcon',
                role: 'Vice President of Operations'
            }, {
                fname: 'Judy',
                lname: 'Lin',
                role: 'Vice President of Engineering'
            }], // data
            sorters: 'lname'
        }); // create()
        
        Ext.create('Ext.List', {
            fullscreen: true,
            store: myStore,
            itemTpl: Ext.create('Ext.XTemplate', '
{lname:this.toUpper}, {fname}
', {
                toUpper: function (item) {
                    return item.toUpperCase();
                } // toUpper()
            }), // itemTpl
            items: [{
                xtype: 'titlebar',
                title: 'Ext.XTemplate custom function',
                docked: 'top'
            }] // items
        }); // create()
    } // launch
}); // application()
