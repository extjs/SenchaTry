Detecting when the user changes an Ext.field.Spinner control in Sencha Touch 2

The following example shows how you detect when the user changes an Ext.field.Spinner (spinnerfield) control in Sencha Touch 2 by listening for the spin event (or the spinup and spindown events separately).

Tags:

  • Sencha Touch
  • Sencha Touch 2.0.1
[example]
© 2012 Sencha Inc. 
/*global Ext:false */
Ext.application({
    launch: function () {
        var tmpl = Ext.create('Ext.XTemplate', '[{evt}] value: {value}, direction: \'{dir}\'');
        
        Ext.create('Ext.form.Panel', {
            fullscreen: true,
            items: [{
                xtype: 'fieldset',
                id: 'fset',
                items: [{
                    xtype: 'spinnerfield',
                    label: 'value:',
                    increment: 1,
                    listeners: {
                        spin: function (spinner, value, dir) {
                            var str = tmpl.apply({evt:'spin', value:value, dir:dir});
                            spinner.up('fieldset').setInstructions(str);
                        } // spin
                    } // listeners
                }] // items (fieldset)
            }] // items (formpanel)
        }); // create()
    } // launch
}); // application()
