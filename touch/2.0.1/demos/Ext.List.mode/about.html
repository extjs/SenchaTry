Allowing multiple selection on an Ext.List component in Sencha Touch 2

The following example shows how you can enable multiple item selection on an Ext.List component in Sencha Touch 2 by setting the mode config option.

Tags: Sencha Touch Sencha Touch 2.0.1
[example]
© 2012 Sencha Inc. 
/*global Ext:false */
Ext.application({
    launch: function () {
        Ext.create('Ext.List', {
            fullscreen: true,
            mode: 'MULTI',
            store: {
                fields: ['name'],
                data: [{
                    name: 'Red'
                }, {
                    name: 'Orange'
                }, {
                    name: 'Yellow'
                }, {
                    name: 'Green'
                }, {
                    name: 'Blue'
                }] // data
            }, // store
            itemTpl: '{name}',
            listeners: {
                selectionchange: function (list, records) {
                    var names = [];
                    Ext.Array.each(records, function (item) {
                        names.push('
  • ' + item.data.name + '
    • ');
                    }); // each()
                    Ext.Msg.alert('You selected ' + records.length + ' item(s)', '
      ' + names.join('') + '
    ');
                } // selectionchange
            } // listeners
        }); // create()
    } // launch
}); // application()
